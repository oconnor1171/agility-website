from flask import Flask, request, jsonify
from datetime import datetime
from appointment_booker.calendar_writer import create_event_and_log

app = Flask(__name__)

@app.route('/book-appointment', methods=['POST'])
def book():
    data = request.json

    # Extract expected fields from the incoming JSON
    name = data.get("name")
    phone = data.get("phone")
    purpose = data.get("purpose")
    datetime_str = data.get("datetime")  # ISO 8601 string
    caller_number = data.get("caller_number")

    if not all([name, phone, purpose, datetime_str, caller_number]):
        return jsonify({"error": "Missing one or more required fields."}), 400

    try:
        # Parse the appointment datetime
        start = datetime.fromisoformat(datetime_str)
        end = start.replace(minute=start.minute + 60)  # Default 30-min duration, changed to 60

        # Call main handler (calendar + sheets + email)
        result = create_event_and_log(
            name=name,
            phone=phone,
            purpose=purpose,
            start=start,
            end=end,
            caller_number=caller_number
        )

        return jsonify(result)
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == "__main__":
    app.run(debug=True)

# Constants

def _authorize():
    creds = service_account.Credentials.from_service_account_file(
        "credentials.json", scopes=SCOPES)
    return creds

def create_event_and_log(data):
    creds = _authorize()
    service = build("calendar", "v3", credentials=creds)

    event = {
        "summary": data["purpose"],
        "description": f"Caller: {data['name']} - {data['phone']}",
        "start": {"dateTime": data["datetime"], "timeZone": TIMEZONE},
        "end": {
            "dateTime": (
                datetime.datetime.fromisoformat(data["datetime"]) +
                datetime.timedelta(hours=1)
            ).isoformat(),
            "timeZone": TIMEZONE,
        },
    }

    event_result = service.events().insert(calendarId="primary", body=event).execute()
    event_id = event_result["id"]

    log_appointment_to_sheets(
        data["datetime"],
        event_result["end"]["dateTime"],
        data["name"],
        data["phone"],
        data["purpose"],
        event_id,
        data["caller_number"]
    )

    return {
        "success": True,
        "message": f"Appointment booked at {data['datetime']}",
        "event_id": event_id,
    }

def cancel_event(event_id):
    creds = _authorize()
    service = build("calendar", "v3", credentials=creds)
    try:
        service.events().delete(calendarId="primary", eventId=event_id).execute()
        log_audit_event("Cancelled", {"event_id": event_id})
        return {"success": True, "message": f"Event {event_id} cancelled."}
    except Exception as e:
        return {"success": False, "error": str(e)}

def cancel_event_by_lookup(phone, start_time):
    creds = _authorize()
    row_data = find_row_by_phone_and_time(creds, phone, start_time)
    if not row_data:
        return {"success": False, "message": "Appointment not found."}

    event_id = row_data.get("event_id")
    if not event_id:
        return {"success": False, "message": "Event ID missing in sheet."}

    try:
        service = build("calendar", "v3", credentials=creds)
        service.events().delete(calendarId="primary", eventId=event_id).execute()
        mark_row_as_cancelled(creds, row_data["row_number"])
        log_audit_event("Cancelled", {"event_id": event_id, "phone": phone, "start_time": start_time})
        return {"success": True, "message": "Appointment cancelled."}
    except Exception as e:
        return {"success": False, "error": str(e)}

def reschedule_event(event_id, new_start_str, new_end_str):

    service = build("calendar", "v3", credentials=creds)
    try:
        event = service.events().get(calendarId="primary", eventId=event_id).execute()
        event["start"]["dateTime"] = new_start_str

        log_audit_event("Rescheduled", {"event_id": event_id, "new_start": new_start_str})
        return {"success": True, "message": "Appointment rescheduled", "event_id": event_id}
    except Exception as e:
        return {"success": False, "error": str(e)}

import smtplib
import os
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

# ✅ Use the exact env var keys from Render
SMTP_USERNAME = os.environ.get("SMTP_USERNAME")
SMTP_PASSWORD = os.environ.get("SMTP_PASSWORD")
CALENDAR_OWNER_EMAIL = os.environ.get("CALENDAR_OWNER_EMAIL", SMTP_USERNAME)

SMTP_SERVER = "smtp.gmail.com"
SMTP_PORT = 587

# ✅ Debug Environment Variables
print(f"[DEBUG] SMTP_USERNAME: {SMTP_USERNAME}")
print(f"[DEBUG] SMTP_PASSWORD present: {bool(SMTP_PASSWORD)}")
print(f"[DEBUG] SMTP_PASSWORD length: {len(SMTP_PASSWORD) if SMTP_PASSWORD else 'None'}")
print(f"[DEBUG] SMTP_PASSWORD valid (16 chars): {len(SMTP_PASSWORD.strip()) == 16 if SMTP_PASSWORD else False}")
print(f"[DEBUG] CALENDAR_OWNER_EMAIL: {CALENDAR_OWNER_EMAIL}")

def send_notification(name, phone, purpose, dt):
    subject = "New Appointment Scheduled"
    body = f"""
    A new appointment has been booked.

    Name: {name}
    Phone: {phone}
    Purpose: {purpose}
    Time: {dt.strftime('%A, %B %d, %Y at %I:%M %p')}
    """

    print(f"[Mailer] Preparing to send email to: {CALENDAR_OWNER_EMAIL}")

    msg = MIMEMultipart()
    msg['From'] = SMTP_USERNAME
    msg['To'] = CALENDAR_OWNER_EMAIL
    msg['Subject'] = subject
    msg.attach(MIMEText(body, 'plain'))

    try:
        with smtplib.SMTP(SMTP_SERVER, SMTP_PORT) as server:
            server.set_debuglevel(1)  # SMTP protocol debug
            server.starttls()
            print("[Mailer] Attempting SMTP login...")
            server.login(SMTP_USERNAME, SMTP_PASSWORD)
            server.sendmail(SMTP_USERNAME, CALENDAR_OWNER_EMAIL, msg.as_string())
            print("[Mailer] Email sent successfully.")
    except smtplib.SMTPAuthenticationError as e:
        print(f"[Mailer] SMTP Authentication Error: {e.smtp_code} - {e.smtp_error.decode()}")
        raise
    except Exception as e:
        print(f"[Mailer] General email error: {e}")
        raise

SPREADSHEET_ID = "1I-OOXkme0VTcnFC-B9LhD-s1Qj3Ki9Q1TPDQ8C663WY"
SALES_SHEET_NAME = "Appointments"
AUDIT_SHEET_NAME = "Audit"

def _authorize():
    creds = service_account.Credentials.from_service_account_file(
        'credentials.json', scopes=SCOPES
    )
    return creds

def log_appointment_to_sheets(creds, row_data):
    service = build('sheets', 'v4', credentials=creds)
    sheet = service.spreadsheets()
    values = [list(row_data.values())]
    body = {'values': values}
    result = sheet.values().append(
        spreadsheetId=SPREADSHEET_ID,
        range=f"{SALES_SHEET_NAME}!A1",
        valueInputOption="USER_ENTERED",
        insertDataOption="INSERT_ROWS",
        body=body
    ).execute()
    return result

def list_all_appointments(creds):
    service = build('sheets', 'v4', credentials=creds)
    result = service.spreadsheets().values().get(
        spreadsheetId=SPREADSHEET_ID,
        range=f"{SALES_SHEET_NAME}!A2:Z"
    ).execute()
    return result.get("values", [])

def update_event_in_sheets(creds, event_id, new_start, new_end, timestamp):
    service = build('sheets', 'v4', credentials=creds)
    sheet = service.spreadsheets()
    result = sheet.values().get(
        spreadsheetId=SPREADSHEET_ID,
        range=f"{SALES_SHEET_NAME}!A2:Z"
    ).execute()
    rows = result.get("values", [])

    for idx, row in enumerate(rows, start=2):
        if len(row) > 0 and row[0] == event_id:
            if len(row) < 11:
                row += [""] * (11 - len(row))
            row[3] = new_start
            row[4] = new_end
            row[10] = timestamp

            update_range = f"{SALES_SHEET_NAME}!A{idx}:K{idx}"
            body = {'values': [row[:11]]}
            sheet.values().update(
                spreadsheetId=SPREADSHEET_ID,
                range=update_range,
                valueInputOption="USER_ENTERED",
                body=body
            ).execute()
            return f"Row {idx} updated with new time."
    return "Event ID not found in sheet."

def find_event_by_phone_and_start(creds, phone, start_time):
    service = build('sheets', 'v4', credentials=creds)
    result = service.spreadsheets().values().get(
        spreadsheetId=SPREADSHEET_ID,
        range=f"{SALES_SHEET_NAME}!A2:Z"
    ).execute()
    rows = result.get("values", [])
    for row in rows:
        if len(row) >= 5:
            event_id = row[0]
            row_start = row[3]
            row_phone = row[2]
            if row_start == start_time and row_phone == phone:
                return event_id
    return None

def cancel_event_in_sheets(creds, event_id, timestamp):
    service = build('sheets', 'v4', credentials=creds)
    sheet = service.spreadsheets()

@app.route("/check-availability", methods=["POST"])
def availability():
    try:
        data = request.get_json()
        if not data or "datetime" not in data:
            return jsonify({"error": "Missing 'datetime' in request."}), 400

        dt = datetime.datetime.fromisoformat(data["datetime"])
        tz = pytz.timezone(TIMEZONE)
        dt_local = dt.astimezone(tz)

        if is_weekend(dt_local):
            return jsonify({
                "available": False,
                "message": "We do not schedule appointments on weekends. Please choose a weekday."
            }), 200

        holiday = is_us_holiday(dt_local)
        if holiday:
            return jsonify({
                "available": False,
                "message": f"We are closed on {holiday}. Please choose another date."
            }), 200

        result = check_availability(data["datetime"])
        return jsonify(result), 200

    except Exception as e:
        return jsonify({"error": str(e)}), 500
